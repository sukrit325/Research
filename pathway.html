# Can you write a function to normalize a wavefunction?
import numpy as np

def normalize_psi(x, psi):
    """Normalize wavefunction psi over domain x"""
    integral = np.trapz(np.abs(psi)**2, x)
    return psi / np.sqrt(integral)

Monte Carlo Methods

Statistical sampling, Metropolis algorithm, phase space exploration.

Statistical Mechanics

Ensembles, partition functions, free energy calculations. Tuckerman's text recommended [^2^]

🎯 Focus Area: DFT

90% of modern computational chemistry uses DFT. Master these concepts:

  • B3LYP is outdated for transition metals; use range-separated hybrids
  • Dispersion corrections (D3, D4) are essential for non-covalent interactions
  • Basis set convergence: triple-zeta minimum for publication-quality results

# Standard workflow for ORCA automation
import os
import subprocess
import cclib

# 1. Generate input files programmatically
def write_orca_input(molecule_name, coords, method="B3LYP", basis="def2-SVP"):
    inp = f"""! {method} {basis} Opt Freq
%pal nprocs 8 end
%maxcore 4000
* xyz 0 1
{coords}
*"""
    with open(f"{molecule_name}.inp", "w") as f:
        f.write(inp)

# 2. Run calculation
subprocess.run(["orca", "molecule.inp"], capture_output=True)

# 3. Parse results with cclib
data = cclib.io.ccread("molecule.out")
print(f"Energy: {data.scfenergies[-1]} eV")
print(f"Dipole moment: {data.moments[1]}")
